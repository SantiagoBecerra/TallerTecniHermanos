drop table USUARIOS cascade constraints;
drop table TIPO_DOCUMENTO cascade constraints;

CREATE TABLE TIPO_DOCUMENTO (
  id_tipo_documento number NOT NULL,
  nombre_documento varchar2(100) NOT NULL,
CONSTRAINT DIST_PK_ID PRIMARY KEY (id_tipo_documento)
);

CREATE TABLE USUARIOS(
  tipo_documento number NOT NULL,
  numero_documento number NOT NULL,
  primer_apellido varchar2(30) NOT NULL,
  segundo_apellido varchar2(30) NOT NULL,
  nombres varchar2(50) NOT NULL,
  fecha_nacimiento DATE NOT NULL,
  telefono number NOT NULL,
  ciudad varchar2(50) NOT NULL,
  direccion varchar2(60) NOT NULL,
  usuario varchar2(20) NOT NULL,
  contrasenia varchar2(20) NOT NULL,
  tipo_usuario char(2) NOT NULL, 
CONSTRAINT PRO_PK_ID PRIMARY KEY (numero_documento),
CONSTRAINT USU_FK_TI_DOC FOREIGN KEY (tipo_documento) REFERENCES TIPO_DOCUMENTO(id_tipo_documento)
);


CREATE TABLE ASIGNATURA(
  nombre varchar2(50) NOT NULL,
  descripcion varchar2(300) NOT NULL,
	CONSTRAINT ASIG_PK_ID PRIMARY KEY (nombre)
);

CREATE TABLE AREA(
  nombre varchar2(50) NOT NULL,
  descripcion varchar2(300) NOT NULL,
	CONSTRAINT ARE_PK_ID PRIMARY KEY (nombre)
);
